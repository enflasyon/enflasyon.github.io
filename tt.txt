import pandas as pd
import plotly.graph_objects as go

# CSV dosyanı oku
df = pd.read_csv("mission_data.csv")

fig = go.Figure()

# 12 adet Time-Mission çifti için
for i in range(1, 13):
    time_col = f"Time{i}"
    mission_col = f"Mission {i}"

    if time_col in df.columns and mission_col in df.columns:
        temp = df[[time_col, mission_col]].dropna()

        fig.add_trace(
            go.Scatter(
                x=temp[time_col],
                y=temp[mission_col],
                mode="lines",
                name=f"Mission {i}"
            )
        )

fig.update_layout(
    title="Mission Profiles",
    xaxis_title="Time",
    yaxis_title="Mission Value",
    template="plotly_white",
    hovermode="x unified",
    width=1200,
    height=700
)

fig.show()
