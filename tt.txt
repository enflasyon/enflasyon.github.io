import pdfplumber
import fitz  # PyMuPDF
import pandas as pd
import re
from pdf2image import convert_from_path
import pytesseract

PDF_PATH = "organizasyon.pdf"
START_PAGE = 55
END_PAGE = 88

TITLE_KEYWORDS = [
    "Mühendis",
    "Uzman Mühendis",
    "Kıdemli Mühendis"
]

name_pattern = re.compile(r"([A-ZÇĞİÖŞÜ][a-zçğıöşü]+(?:\s[A-ZÇĞİÖŞÜ][a-zçğıöşü]+){1,2})")

results = []

def extract_from_text():
    with pdfplumber.open(PDF_PATH) as pdf:
        for i in range(START_PAGE - 1, END_PAGE):
            text = pdf.pages[i].extract_text()
            if not text:
                continue

            for line in text.split("\n"):
                name_match = name_pattern.search(line)
                for title in TITLE_KEYWORDS:
                    if title in line and name_match:
                        results.append({
                            "page": i + 1,
                            "name": name_match.group(1),
                            "title": title
                        })

def extract_with_ocr():
    images = convert_from_path(
        PDF_PATH,
        first_page=START_PAGE,
        last_page=END_PAGE,
        dpi=300
    )

    for idx, img in enumerate(images):
        text = pytesseract.image_to_string(img, lang="tur")
        for line in text.split("\n"):
            name_match = name_pattern.search(line)
            for title in TITLE_KEYWORDS:
                if title in line and name_match:
                    results.append({
                        "page": START_PAGE + idx,
                        "name": name_match.group(1),
                        "title": title
                    })

# Önce text dene
extract_from_text()

# Eğer hiç sonuç yoksa OCR’a düş
if len(results) == 0:
    extract_with_ocr()

df = pd.DataFrame(results)
df.to_csv("organizasyon_55_88.csv", index=False, encoding="utf-8-sig")

print("CSV üretildi:", len(df), "kayıt")
