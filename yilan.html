from pathlib import Path from tkinter import Tk, filedialog, messagebox from pypdf import PdfReader, PdfWriter OUTPUT_FILENAME = "Birlesik.pdf" def merge_pdfs(folder_path: Path) -> Path: output_path = folder_path / OUTPUT_FILENAME # Sadece seçilen klasörün içindeki PDF dosyalarını alır. # Daha önce oluşturulan çıktı dosyasını tekrar birleştirmez. pdf_files = sorted( ( file for file in folder_path.iterdir() if file.is_file() and file.suffix.lower() == ".pdf" and file.name.lower() != OUTPUT_FILENAME.lower() ), key=lambda file: file.name.lower(), ) if not pdf_files: raise FileNotFoundError("Seçilen klasörde birleştirilecek PDF bulunamadı.") writer = PdfWriter() added_files = [] try: for pdf_file in pdf_files: try: reader = PdfReader(str(pdf_file)) if reader.is_encrypted: try: reader.decrypt("") except Exception: print(f"Şifreli PDF atlandı: {pdf_file.name}") continue for page in reader.pages: writer.add_page(page) added_files.append(pdf_file.name) print(f"Eklendi: {pdf_file.name}") except Exception as error: print(f"PDF okunamadı, atlandı: {pdf_file.name}") print(f"Hata: {error}") if not added_files: raise RuntimeError("Hiçbir PDF dosyası başarıyla birleştirilemedi.") with output_path.open("wb") as output_file: writer.write(output_file) finally: writer.close() return output_path def main() -> None: root = Tk() root.withdraw() folder = filedialog.askdirectory( title="Birleştirilecek PDF dosyalarının bulunduğu klasörü seçin" ) if not folder: return try: output_path = merge_pdfs(Path(folder)) messagebox.showinfo( "İşlem tamamlandı", f"PDF dosyaları başarıyla birleştirildi.\n\n" f"Çıktı:\n{output_path}", ) except Exception as error: messagebox.showerror( "Hata", str(error), ) finally: root.destroy() if __name__ == "__main__": main()
